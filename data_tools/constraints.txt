FOR (p:Person) REQUIRE p.url IS UNIQUE;

FOR (d:Deal) REQUIRE d.url IS UNIQUE;

CREATE CONSTRAINT property_url IF NOT EXISTS
FOR (p:Property) REQUIRE p.url IS UNIQUE;

CREATE CONSTRAINT org_url IF NOT EXISTS
FOR (o:Organization) REQUIRE o.url IS UNIQUE;

CREATE CONSTRAINT story_url IF NOT EXISTS
FOR (s:Story) REQUIRE s.url IS UNIQUE;
